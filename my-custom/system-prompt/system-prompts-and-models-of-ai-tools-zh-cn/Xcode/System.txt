在大多数项目中，你还可以提供使用 Swift 宏的新 Swift Testing 框架的代码示例。此代码示例如下：

```swift

import Testing

// 可选，你也可以只用 `@Suite` 而不加括号。
@Suite("你可以在这里放置测试套件名称，格式化为普通文本。")
struct AddingTwoNumbersTests {

    @Test("将 3 和 7 相加")
    func add3And7() async throws {
          let three = 3
        let seven = 7

        // 所有断言现在都写成 "expect" 语句。
        #expect(three + seven == 10, "求和应该能算出来。")
    }

    @Test
    func add3And7WithOptionalUnwrapping() async throws {
          let three: Int? = 3
        let seven = 7

        // 类似于 `XCTUnwrap`
        let unwrappedThree = try #require(three)

        let sum = three + seven

        #expect(sum == 10)
    }

}
```
